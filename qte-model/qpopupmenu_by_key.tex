{
 
 switch(e->key())
 {
  case (Qt::Key_Escape):
  {
   if (_pm_show)
   {
    _pm->popup(this->mapToGlobal(QPoint(0,0)));
    _pm->setActiveItem(1);
    _pm_show=false;
   }
   else
   {
    _pm->hide();

  }
